/**
 * 모드 전역에서 사용하는 모든 레지스트리(Block, Item, CreativeTab 등)를 선언하고
 * 이벤트 버스에 등록하는 핵심 구성 클래스입니다.
 *
 * - BLOCKS / ITEMS : 블록·아이템 레지스트리
 * - CREATIVE_MODE_TABS : 크리에이티브 탭 레지스트리
 * - register(...) : 실제 레지스트리를 이벤트 버스(FML Mod Event Bus)에 등록
 */
public final class ModContent {

    /** 블록 등록기 */
    public static final DeferredRegister.Blocks BLOCKS =
            DeferredRegister.createBlocks(ExampleMod.MODID);

    /** 아이템 등록기 */
    public static final DeferredRegister.Items ITEMS =
            DeferredRegister.createItems(ExampleMod.MODID);

    /** 크리에이티브 탭 등록기 */
    public static final DeferredRegister CREATIVE_MODE_TABS =
            DeferredRegister.create(Registries.CREATIVE_MODE_TAB, ExampleMod.MODID);

    /**
     * 예제 블록 등록
     * - 단순한 스톤 색상의 블록
     */
    public static final DeferredBlock EXAMPLE_BLOCK =
            BLOCKS.registerSimpleBlock(
                    "example_block",
                    BlockBehaviour.Properties.of().mapColor(MapColor.STONE)
            );

    /**
     * 예제 블록의 BlockItem 등록
     * - 블록을 인벤토리에서 들 수 있게 하는 아이템
     */
    public static final DeferredItem EXAMPLE_BLOCK_ITEM =
            ITEMS.registerSimpleBlockItem("example_block", EXAMPLE_BLOCK);

    /**
     * 예제 아이템 등록
     * - 간단한 식용 아이템
     */
    public static final DeferredItem EXAMPLE_ITEM =
            ITEMS.registerSimpleItem(
                    "example_item",
                    new Item.Properties().food(
                            new FoodProperties.Builder()
                                    .alwaysEdible()     // 언제나 먹기 가능
                                    .nutrition(1)       // 배고픔 회복량
                                    .saturationModifier(2f) // 포화도 계수
                                    .build()
                    )
            );

    /**
     * 예제 크리에이티브 탭 등록
     * - 탭 아이콘 : example_item
     * - 탭 위치 : 전투 탭 앞
     * - 탭 내부 아이템 : example_item
     */
    public static final DeferredHolder EXAMPLE_TAB =
            CREATIVE_MODE_TABS.register(
                    "example_tab",
                    () -> CreativeModeTab.builder()
                            .title(Component.translatable("itemGroup.examplemod"))
                            .withTabsBefore(CreativeModeTabs.COMBAT)
                            .icon(() -> EXAMPLE_ITEM.get().getDefaultInstance())
                            .displayItems((parameters, output) ->
                                    output.accept(EXAMPLE_ITEM.get()))
                            .build()
            );

    /**
     * StudyBlockRegistry 내부의 블록 초기화 작업을 먼저 수행합니다.
     * (별도로 만든 학습용 블록 레지스트리 코드 연결)
     */
    static {
        StudyBlockRegistry.bootstrap();
    }

    private ModContent() {
    }

    /**
     * 실제 등록 단계
     * - FML Mod Event Bus 에 연결해야 레지스트리가 적용됩니다.
     */
    public static void register(IEventBus eventBus) {
        BLOCKS.register(eventBus);
        ITEMS.register(eventBus);
        CREATIVE_MODE_TABS.register(eventBus);
    }
}

01. 페이지 개요

이 페이지는 NeoForge 1.21.1 기반 모드 개발 과정을 자유로운 레이아웃으로 기록하기 위한 서브 페이지입니다. 이미지, 영상, 로그, 코드 일부 등을 섹션별로 정리해 둘 수 있도록 구성했습니다.

환경 구축 HUD 튜닝 네트워크 UI/UX

02. 핵심 하이라이트

Dev 환경 & 빌드 파이프라인

gradle 빌드 시 클라이언트/서버 모드 폴더로 자동 배포되도록 설정한 내용, 버전 관리 전략 등을 요약할 수 있는 공간입니다.

커스텀 HUD & 스프라이트

widgets 시트 분해, 스프라이트 크기 조정, 기존 HUD와의 공존 방식을 시각적으로 설명할 수 있는 카드입니다.

네트워크 & 동기화

money, 여권 UI, 서버-클라 동기 로직 등 통신 계층을 요약해서 적어 둘 수 있습니다.

04. HUD 전체 레이아웃 스냅샷

체력/허기/경험치/레벨/커스텀 영역을 한 장에 담아서 변경 전/후를 비교할 수 있는 섹션입니다. 이미지는 자유롭게 교체하시면 됩니다.

예시로 placeholder 이미지를 사용하고 있으니, 실제 과제에서 캡처한 화면으로 교체해 주세요.

05. Sprite 실험 로그

이미지 크기 차이, 자르기 방식, 레티나/픽셀 아트 스타일 차이 등 실제로 겪었던 시행착오를 자유롭게 적을 수 있는 자리입니다.

  • 기본 widget 시트 크기 vs 커스텀 이미지 크기 비교
  • 좌표 기준(픽셀)과 논리 좌표 구분
  • HUD가 잘리는 문제 / 여백 처리 등

06. 플레이 영상

HUD / money / 여권 UI가 실제로 동작하는 장면을 담은 영상입니다. 필요하면 여러 버전을 서로 다른 구간에 배치할 수도 있습니다.
